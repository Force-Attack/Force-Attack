# The Rwandan Revolution
country_event = {
	id = ww_congo.20
	title = ww_congo.20.t
	desc = ww_congo.20.d
	picture = GFX_report_event_gathering_protest
	
	is_triggered_only = yes

	fire_only_once = yes

	immediate = {
		hidden_effect = {
			set_global_flag = rwandan_revolution_happened_flag
			768 = {	
				transfer_state_to = RWA
			}
			RWA = {
				delete_unit_template_and_units = {
					division_template = "Mwami's Guard"
					disband = yes #if yes, will refund equipment/manpower.
				}

				division_template = {
					name = "Mwami's Guard"		
					division_names_group = FREN_INFB_01		# Uses French-language generic division names
					template_counter = 67
					regiments = {
						irregular_infantry = { x = 0 y = 0 }
						irregular_infantry = { x = 0 y = 1 }
						irregular_infantry = { x = 0 y = 2 }
						irregular_infantry = { x = 1 y = 0 }
						irregular_infantry = { x = 1 y = 1 }
					}
					support = {
					}
				}

				random_state = { 
					limit = {
						is_owned_by = PREV
					}
					create_unit = {
					    division = "name = \"Mwami's First Guard\" division_template = \"Mwami's Guard\" start_experience_factor = 0.4 start_equipment_factor = 0.6"
					    owner = PREV
					}
					create_unit = {
					    division = "name = \"Mwami's Second Guard\" division_template = \"Mwami's Guard\" start_experience_factor = 0.2 start_equipment_factor = 0.5"
					    owner = PREV
					}
				}
			}
		}
	}
	
	# It is not our place to stop them
	option = {
		name = ww_congo.20.a
		ai_chance = {
			base = 80
		}
		
		add_political_power = 50
		add_stability = 0.02

		IF = {
			limit = {
				768 = {
					is_core_of = ROOT
				}
			}
			768 = {
				remove_core_of = ROOT 
			}
		}
		IF = {
			limit = {
				768 = {
					is_claimed_by = ROOT
				}
			}
			768 = {
				remove_claim_by = ROOT 
			}
		}

	}

	# Who said they could have independence?
	option = {
		name = ww_congo.20.b
		ai_chance = {
			base = 20
		}
		
		IF = {
			limit = {
				768 = {
					NOT = {
						is_claimed_by = ROOT
					}
				}
			}
			768 = {
				add_claim_by = ROOT 
			}
		}

		declare_war_on = {
			target = RWA
			type = annex_everything
		}	

	}
