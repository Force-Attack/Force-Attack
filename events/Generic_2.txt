# COUNTRY INFLUENCING OUR POLITICS
country_event = {
	id = ai_ideology.1
	title = generic.1.t
	desc = { 
		text = generic.1.d_neutral_good
	}
	picture = GFX_report_event_generic_read_write

	is_triggered_only = yes

	option = {
		name = generic.1.a
		if = {
			limit = {
				NOT = {
					has_government = fascism
				}
			}
			set_politics = {
			    ruling_party = fascism
			    elections_allowed = no
			}
		}
	}
}
