add_namespace = my_mod

country_event = {
    id = my_mod.1
    title = "Special Forces Over Cap!"
    desc = "Our military has exceeded the allowed number of special forces. We are suffering penalties due to overburdened logistics and training inefficiencies."
    picture = GFX_report_major_battle

    trigger = {
        NOT = { has_country_flag = special_forces_penalized_flag }  # Avoid firing if penalty is already applied
        special_forces_over_cap = yes  # Check if the country has more than 50 special forces divisions
    }

    immediate = {
        add_dynamic_modifier = {
            modifier = special_forces_over_50_cap
            days = -1
        }
        set_country_flag = special_forces_penalized_flag
		log = "Event 1: Special Forces Over Cap Triggered"
    }

    option = {
        name = "We must fix this!"
    }
}

country_event = {
    id = my_mod.2
    title = "Special Forces Below Cap"
    desc = "Our special forces are now within acceptable limits, and the penalties have been lifted."
    picture = GFX_report_major_battle

    trigger = {
        has_country_flag = special_forces_penalized_flag
        NOT = { special_forces_over_cap = yes }  # Only fire when the country has 50 or fewer special forces divisions
    }

    immediate = {
        remove_dynamic_modifier = special_forces_over_50_cap
        clr_country_flag = special_forces_penalized_flag
    }

    option = {
        name = "Good!"
    }
}
