AUS_loans_up = {
	if = {
		limit = { 
			NOT = {
				has_idea = AUS_reliant_on_lon_1
				has_idea = AUS_reliant_on_lon_2
				has_idea = AUS_reliant_on_lon_3
				has_idea = AUS_reliant_on_lon_4
			}
		}
		add_ideas = AUS_reliant_on_lon_1
	}
	else_if = {
		limit = {
			has_idea = AUS_reliant_on_lon_1
		}
		swap_ideas = {
			remove_idea = AUS_reliant_on_lon_1
			add_idea = AUS_reliant_on_lon_2
		}
	}
	else_if = {
		limit = {
			has_idea = AUS_reliant_on_lon_2
		}
		swap_ideas = {
			remove_idea = AUS_reliant_on_lon_2
			add_idea = AUS_reliant_on_lon_3
		}
	}
	else_if = {
		limit = {
			has_idea = AUS_reliant_on_lon_3
		}
		swap_ideas = {
			remove_idea = AUS_reliant_on_lon_3
			add_idea = AUS_reliant_on_lon_4
		}
	}
}
