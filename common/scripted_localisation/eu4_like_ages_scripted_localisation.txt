defined_text = {
	name = GetNewAgePopupUniqueAgeAbility1

	text = {
		trigger = {
			if = {
				limit = {
					check_variable = { global.current_age = 1 }
				}
				set_temp_variable = { age_ability = 41 }
			}
			else_if = {
				limit = {
					check_variable = { global.current_age = 2 }
				}
				set_temp_variable = { age_ability = 52 }
			}
			else_if = {
				limit = {
					check_variable = { global.current_age = 3 }
				}
				set_temp_variable = { age_ability = 8 }
			}
			else_if = {
				limit = {
					check_variable = { global.current_age = 4 }
				}
				set_temp_variable = { age_ability = 19 }
			}
			else_if = {
				limit = {
					check_variable = { global.current_age = 5 }
				}
				set_temp_variable = { age_ability = 30 }
			}
			else_if = {
				limit = {
					check_variable = { global.current_age = 6 }
				}
				set_temp_variable = { age_ability = 63 }
			}
			always = yes
		}
		localization_key = "[This.GetAgeAbilityOtherContext]"
	}
}
